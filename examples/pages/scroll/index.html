滚动容器


示例

条目 {{i}}

代码

    
      direction // {String} 滚动方向，默认水平方向。可选值：horizontal | vertical
      autoCenter // {Boolean} 是否开启自动居中功能，暂不支持竖直方向
      elSelector // {Boolean} 需要居中的元素选择器
      dataModel //  {Number} 数据模型，修改此值即可重新计算，适用于子元素变化等场景

代码

    
      <fr-scroll
          :auto-center="true"
          :el-selector="elSelector">

      <span class="scroll-item"
               :class="'scroll-item-' + i"
               @click="centerEle('.scroll-item-' + i)"
               v-for="i in 10">
               条目文案
      </span>
      </fr-scroll>
    
  
    
      new Vue({
        el: '#app',
        data() {
          return {
            elSelector: '',
          }
        },
        methods: {
          centerEle(selector) {
            this.elSelector = selector;
            this.$nextTick(() => this.elSelector = '');
          }
        },
      });
